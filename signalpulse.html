    async function fetchHistoricalData() {

        try {

            const date = new Date();

            date.setDate(date.getDate() - historicalDays);

            const startDate = date.toISOString().split('T')[0];

            const url = `https://marketdata.tradermade.com/api/v1/historical?currency=${currencies}&date=${startDate}&api_key=${apiKey}`;

            const response = await fetch(url);

            const data = await response.json();

            if (data && data.quotes) {

                data.quotes.forEach(quote => {

                    const pair = `${quote.base_currency}/${quote.quote_currency}`;

                    if (!historicalData[pair]) {

                        historicalData[pair] = [];

                    }

                    historicalData[pair].push({

                        date: quote.date,

                        mid: quote.mid

                    });

                    // Update price history

                    if (!priceHistory[pair]) {

                        priceHistory[pair] = [];

                    }

                    priceHistory[pair].push(quote.mid);

                });

            } else {

        }

    }

    function calculateEMA(prices, period) {

        const k = 2 / (period + 1);

        let ema = prices.slice(0, period).reduce((sum, price) => sum + price, 0) / period;

        for (let i = period; i < prices.length; i++) {

            ema = (prices[i] * k) + (ema * (1 - k));

        }

        return ema;

    }

    function calculateRSI(prices, period) {

        let gains = 0;

        let losses = 0;

        for (let i = 1; i < period; i++) {

            const change = prices[i] - prices[i - 1];

            if (change > 0) {

                gains += change;

            } else {

                losses -= change;

            }

        }

        gains /= period;

        losses /= period;

        const rs = gains / losses;

        return 100 - (100 / (1 + rs));

    }

    function generateSignal(price, pair) {

        const prices = priceHistory[pair] || [];

        // Ensure there are enough prices for calculations

        if (prices.length < 14) {

            return 'Hold';

        }

        // Calculate EMA and RSI

        const emaPeriod = 14;

        const rsiPeriod = 14;

        const ema = calculateEMA(prices, emaPeriod);

        const rsi = calculateRSI(prices, rsiPeriod);

        const emaSignal = price > ema ? 'Buy' : price < ema ? 'Sell' : 'Hold';

        const rsiSignal = rsi > 70 ? 'Sell' : rsi < 30 ? 'Buy' : 'Hold';

        // Combining signals from EMA and RSI

        if (emaSignal === 'Buy' && rsiSignal === 'Buy') {

            return 'Buy';

        } else if (emaSignal === 'Sell' && rsiSignal === 'Sell') {

            return 'Sell';

        } else {

            return 'Hold';

        }

    }

        function calculatePositionSize() {

            const accountBalance = parseFloat(document.getElementById('accountBalance').value);

                return;

            }

            // Convert stop loss pips to decimal

            const stopLossDecimal = stopLossPips * 0.0001;

            // Calculate position size in units

            const riskAmount = (riskPercentage/ 100) * accountBalance;

            const positionSizeUnits = riskAmount / stopLossDecimal;

            // Convert position size to standard, mini, and micro lots

            const standardLots = positionSizeUnits / 100000;

            const miniLots = standardLots * 10;

            const microLots = standardLots * 100;

            // Display the results in a clear format

                <p>Mini Lots: ${miniLots.toFixed(2)} lots</p>

                <p>Micro Lots: ${microLots.toFixed(2)} lots</p>

            `;

        }
